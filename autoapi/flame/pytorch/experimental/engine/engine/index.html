Process:

call setup(data, max_epochs, max_iterations, epoch_length)
call start_epoch_loop() as
    while not done (epoch < max_epochs):
        call step_epoch() as
            update epoch
            call start_iteration_loop() as
                update iteration
                while True:
                    try:
                        call step_iteration()
                    except StopIteration:
                        break
